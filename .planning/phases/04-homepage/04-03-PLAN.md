```typescript
import { setRequestLocale } from 'next-intl/server';
import { getTranslations } from 'next-intl/server';
import { Hero } from '@/components/sections/hero';
import { HowIWork } from '@/components/sections/how-i-work';
import { Timeline } from '@/components/sections/timeline';
import { Metrics } from '@/components/sections/metrics';
import { CurrentWork } from '@/components/sections/current-work';
import { Contact } from '@/components/sections/contact';

export default async function HomePage({ params }: { params: Promise<{ locale: string }> }) {
  const { locale } = await params;
  setRequestLocale(locale);
  const t = await getTranslations('homepage');

  // Rich text renderer for bold numbers (accent-colored)
  const richText = {
    b: (chunks: React.ReactNode) => <strong className="font-bold text-accent">{chunks}</strong>
  };

  return (
    <>
      <Hero
        headline={t('hero.headline')}
        subtitle={t('hero.subtitle')}
        primaryCta={t('hero.primaryCta')}
        primaryCtaHref="#contact"
        secondaryCta={t('hero.secondaryCta')}
        secondaryCtaHref={locale === 'it' ? '/it/work' : '/work'}
        tags={[t('hero.tag1'), t('hero.tag2'), t('hero.tag3'), t('hero.tag4')]}
      />
      <HowIWork
        label={t('howIWork.label')}
        title={t('howIWork.title')}
        intro={t('howIWork.intro')}
        pillars={[
          { title: t('howIWork.pillar1.title'), text: t.rich('howIWork.pillar1.text', richText) },
          { title: t('howIWork.pillar2.title'), text: t.rich('howIWork.pillar2.text', richText) },
          { title: t('howIWork.pillar3.title'), text: t.rich('howIWork.pillar3.text', richText) },
        ]}
      />
      <Timeline
        label={t('timeline.label')}
        title={t('timeline.title')}
        blocks={[
          { role: t('timeline.block1.role'), company: t('timeline.block1.company'), dates: t('timeline.block1.dates'), text: t.rich('timeline.block1.text', richText) },
          { role: t('timeline.block2.role'), company: t('timeline.block2.company'), dates: t('timeline.block2.dates'), text: t.rich('timeline.block2.text', richText) },
          { role: t('timeline.block3.role'), company: t('timeline.block3.company'), dates: t('timeline.block3.dates'), text: t.rich('timeline.block3.text', richText) },
          { role: t('timeline.block4.role'), company: t('timeline.block4.company'), dates: t('timeline.block4.dates'), text: t.rich('timeline.block4.text', richText) },
        ]}
        closingLine={t('timeline.closingLine')}
      />
      <Metrics
        label={t('metrics.label')}
        title={t('metrics.title')}
        metrics={[
          { number: t('metrics.metric1.number'), context: t('metrics.metric1.context') },
          { number: t('metrics.metric2.number'), context: t('metrics.metric2.context') },
          { number: t('metrics.metric3.number'), context: t('metrics.metric3.context') },
          { number: t('metrics.metric4.number'), context: t('metrics.metric4.context') },
          { number: t('metrics.metric5.number'), context: t('metrics.metric5.context') },
          { number: t('metrics.metric6.number'), context: t('metrics.metric6.context') },
        ]}
        linkText={t('metrics.link')}
        linkHref={locale === 'it' ? '/it/work' : '/work'}
      />
      <CurrentWork
        label={t('currentWork.label')}
        title={t('currentWork.title')}
        cards={[
          { title: t('currentWork.card1.title'), text: t('currentWork.card1.text') },
          { title: t('currentWork.card2.title'), text: t('currentWork.card2.text') },
        ]}
      />
      <Contact
        label={t('contact.label')}
        title={t('contact.title')}
        text={t('contact.text')}
        email={t('contact.email')}
        linkedinLabel={t('contact.linkedinLabel')}
        linkedinHref="https://linkedin.com/in/selfrules"
        cvLabel={t('contact.cvLabel')}
        cvHref="/cv.pdf"
        availability={t('contact.availability')}
      />
    </>
  );
}

export function generateStaticParams() {
  return [{ locale: 'en' }, { locale: 'it' }];
}
```

Key implementation notes:
- `t.rich()` is used for pillar and timeline text that contains `<b>` tags in the JSON. The `richText` object maps `<b>` to `<strong className="font-bold text-accent">`.
- `secondaryCtaHref` and `linkHref` are locale-aware (prefix `/it/` for Italian).
- `linkedinHref` is hardcoded (not locale-dependent).
- `cvHref` points to `/cv.pdf` (placeholder, PDF added in deployment phase).
- `generateStaticParams` ensures both locales are statically generated.
- No 'use client' directive -- this is an async Server Component.
