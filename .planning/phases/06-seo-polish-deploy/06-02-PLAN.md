    1. Create `src/components/ui/scroll-reveal.tsx` (per D-01, D-02, D-03):
       ```typescript
       'use client';

       import { useEffect, useRef, type ReactNode } from 'react';

       export function ScrollReveal({ children }: { children: ReactNode }) {
         const ref = useRef<HTMLDivElement>(null);

         useEffect(() => {
           const el = ref.current;
           if (!el) return;

           // Respect prefers-reduced-motion (D-04, A11Y-05)
           if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) {
             el.classList.remove('animate-fade-up-initial');
             return;
           }

           const observer = new IntersectionObserver(
             ([entry]) => {
               if (entry.isIntersecting) {
                 el.classList.remove('animate-fade-up-initial');
                 el.classList.add('animate-fade-up');
                 observer.unobserve(el);
               }
             },
             { threshold: 0.1 }  // D-03: trigger at 10% visibility
           );

           observer.observe(el);
           return () => observer.disconnect();
         }, []);

         return (
           <div ref={ref} className="animate-fade-up-initial">
             {children}
           </div>
         );
       }
       ```
       - Per D-01: Applied at section level only, NOT individual cards
       - Per D-01: Hero section does NOT use this wrapper (already visible on load)
       - Per D-02: fade-up style (opacity 0->1 + translateY 24px), 600ms ease-out
       - Per D-03: threshold 0.1 (10% visibility trigger)
       - Observer fires once and unobserves
