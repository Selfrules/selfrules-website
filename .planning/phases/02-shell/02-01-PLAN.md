The updated layout should look like:
```typescript
import { setRequestLocale } from 'next-intl/server';
import { NextIntlClientProvider } from 'next-intl';
import { getMessages, getTranslations } from 'next-intl/server';
import { routing } from '@/i18n/routing';
import { notFound } from 'next/navigation';
import { hasLocale } from 'next-intl';
import { Footer } from '@/components/layout/Footer';
import Script from 'next/script';

export function generateStaticParams() {
  return routing.locales.map((locale) => ({ locale }));
}

export default async function LocaleLayout({
  children,
  params,
}: {
  children: React.ReactNode;
  params: Promise<{ locale: string }>;
}) {
  const { locale } = await params;

  if (!hasLocale(routing.locales, locale)) {
    notFound();
  }

  setRequestLocale(locale);
  const messages = await getMessages();
  const t = await getTranslations({ locale, namespace: 'a11y' });

  return (
    <NextIntlClientProvider messages={messages}>
      {/* Skip-to-content link per A11Y-01 */}
      <a
        href="#main"
        className="sr-only focus:not-sr-only focus:fixed focus:top-2 focus:left-2 focus:z-[60] focus:bg-accent focus:text-primary focus:px-4 focus:py-2 focus:text-[length:--text-small] focus:font-medium"
      >
        {t('skipToContent')}
      </a>

      {/* Navbar will be added in Plan 02 */}

      <main id="main" className="pt-[64px]">
        {children}
      </main>

      <Footer locale={locale} />

      {process.env.NEXT_PUBLIC_UMAMI_ID && (
        <Script
          src="/api/umami/script.js"
          data-website-id={process.env.NEXT_PUBLIC_UMAMI_ID}
          strategy="afterInteractive"
        />
      )}
    </NextIntlClientProvider>
  );
}
```

CRITICAL:
- `params` is a Promise in Next.js 16 -- MUST `await params`.
- `setRequestLocale(locale)` MUST be called before any translation hooks.
- `pt-[64px]` on main offsets content below the fixed navbar (64px navbar-height from UI-SPEC).
- Umami script conditionally renders only if `NEXT_PUBLIC_UMAMI_ID` env var is set.
- Skip-to-content uses `sr-only` (hidden) but `focus:not-sr-only focus:fixed` (visible on Tab focus) per A11Y-01.
