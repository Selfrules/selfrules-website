**Step 5: Create src/components/sections/page-cta.tsx**
Per D-10: shared closing CTA for all inner pages. Lighter than homepage Contact section. Props: text (string), primaryCta ({label, href}), secondaryCta? ({label, href}).

```typescript
import { Section } from '@/components/layout/section';
import { Button } from '@/components/ui/button';

interface PageCTAProps {
  text: string;
  primaryCta: { label: string; href: string };
  secondaryCta?: { label: string; href: string };
}

export function PageCTA({ text, primaryCta, secondaryCta }: PageCTAProps) {
  return (
    <Section>
      <div className="mt-[80px]">
        <p className="mb-8 text-base leading-[1.7] text-primary">{text}</p>
        <div className="flex gap-4">
          <Button variant="primary" href={primaryCta.href}>
            {primaryCta.label}
          </Button>
          {secondaryCta && (
            <Button variant="secondary" href={secondaryCta.href}>
              {secondaryCta.label}
            </Button>
          )}
        </div>
      </div>
    </Section>
  );
}
```
